172 void SearchPairHelper::searchTreeBySearchTerm(QString _searchTerm)
173 {
174  if (!_searchedTermList.isEmpty())
175  _searchedTermList.clear();
176  int _totalParents = _tree->model()->rowCount(QModelIndex());
177 
178  for (int _idParent = 0; _idParent < _totalParents; _idParent++)
179  {
180  QModelIndex _idxParent = _tree->model()->index(_idParent,0,QModelIndex());
181  // Getting the Child
182  int _noChildren = _tree->model()->rowCount(_idxParent);
183  for (int _idChild = 0; _idChild < _noChildren; _idChild++)
184  {
185  QModelIndex _idxChildren = _tree->model()->index(_idChild,0,_idxParent);
186  QVariant _dataChild = _idxChildren.data(Qt::DisplayRole);
187  QString _childName = _dataChild.toString();
188  SolverSectionVO _sectionVO = _solverManager->getVariableListofSubsections(_childName);
189  QList<SolverVariableVO> _list = _sectionVO.getSubsectionVarList();
190  QList<int> _resultChildren;
191  for (int i = 0; i< _list.count();i++)
192  {
193  SolverVariableVO _solver = _list.at(i);
194  QString _nameVariable = _solver.getVariableName().toLower();
195  if (_nameVariable.contains(_searchTerm.toLower()))
196  {
197  _resultChildren.append(i);
198  _totalFoundElements ++;
199  }
200  }
201 
202  if (!_resultChildren.isEmpty())
203  {
204  SearchedPair _pairFound;
205  _pairFound.setIdParent(_idParent);
206  _pairFound.setIdChild(_idChild);
207  _pairFound.setChildrenList(_resultChildren);
208  _searchedTermList.append(_pairFound);
209  }
210 
211  }
212 
213  }
