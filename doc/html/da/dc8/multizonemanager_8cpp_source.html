211 bool MultizoneManager::checkProjectBasePathConsistency(MultizoneProjectData * _multizoneProjectData,QString _currentFilePath)
212 {
213  bool _statusNeedToBeSaved = false;
214  QString _projectPath = _multizoneProjectData->getProjectPath();
215  QString _cRootPath = QFileInfo(QFileInfo(_currentFilePath).path()).path();
216  if (_projectPath.isNull())
217  return _statusNeedToBeSaved;
218  QFileInfo _basePathInfo(_projectPath);
219 
220  QString _rootPath = _basePathInfo.path();
221  QString _projectBaseName = _basePathInfo.fileName();
222  if (QString::compare(_rootPath, _cRootPath) != StrEqual)
223  {
224  QString _updatedProjectPath = QDir(_cRootPath).filePath(_projectBaseName);
225  _multizoneProjectData->setProjectPath(_updatedProjectPath);
226 
227  QString _newProjectPathFiles = QDir(_updatedProjectPath).filePath(TOOL4);
228  // Update Configuration Files
229  _multizoneProjectData->setListConfigFiles(updateFilePaths(_multizoneProjectData->getListConfigFileNames(), _newProjectPathFiles));
230  _multizoneProjectData->setListMeshFiles(updateFilePaths(_multizoneProjectData->getListMeshFileNames(), _newProjectPathFiles));
231 
232  _statusNeedToBeSaved = true;
233  }
234  return _statusNeedToBeSaved;
235 }

290 void MultizoneManager::updateMultiZoneValues(QString _filePath, int _option, MultizoneProjectData *_multizoneProjectData,
291  ProjectData * _projectData, bool _isNewConfigFile)
292 {
293  switch(_option)
294  {
295  case OptionMzConfigPath:

299  {
300  int _posToAdd = getPositionToAddFileMultizone(_multizoneProjectData);
301  if (_posToAdd != -1)
302  {
303  if (_posToAdd >= _listConfig.size())
304  _listConfig.append(_filePath);
305  else
306  _listConfig[_posToAdd] = _filePath;
307  _multizoneProjectData->setListConfigFiles(_listConfig);
308  }
309  }
310 
311  }
312  break;
313  case OptionMzMeshPath:
314  {
315  QStringList _listMesh = _multizoneProjectData->getListMeshFiles();
316  int _posToAdd = -1;
317  if (_isNewConfigFile)
318  _posToAdd = getPositionToAddFileMultizone(_multizoneProjectData);
319  else
320  _posToAdd = getConfigurationFilePosition(_multizoneProjectData,_projectData);
321 
322  if (_posToAdd != -1)
323  {
324  if (_posToAdd >= _listMesh.size())
325  _listMesh.append(_filePath);
326  else
327  _listMesh[_posToAdd] = _filePath;
328  _multizoneProjectData->setListMeshFiles(_listMesh);
329  }
330 
331  }
332  break;
333  }

342 int MultizoneManager::getPositionToAddFileMultizone( MultizoneProjectData *_multizoneProjectData)
343 {
344  int _result = -1;

346  QStringList _listMesh = _multizoneProjectData->getListMeshFiles();
347  if (_listConfig.size() == _listMesh.size())
348  return (_listConfig.count());
349
