110 QString SolverVariableVO::getHelpContents(bool _withFormat)
111 {
112  QString _text;
113  if (_withFormat)
114  {
115  HtmlFormatter _htmlFormatter;
116  _htmlFormatter.startFormatting();
117  _htmlFormatter.appendTitle(this->getVariableName());
118  _htmlFormatter.appendTitleParagraph("Definition");
119  _htmlFormatter.appendParagraph(_helpContents->getDefinitionText());
120  _htmlFormatter.appendTitleParagraph("Possible values and default Values");
121  _htmlFormatter.appendParagraph( _helpContents->getValuesText());
122  _htmlFormatter.appendTitleParagraph("Other Details");
123  _htmlFormatter.appendParagraph(_helpContents->getDetailsText());
124 
125  // Adding the values in a table
126  _htmlFormatter.appendTitleParagraph("Quick Summary");
127  QStringList _headerTable;
128  QList<QStringList> _contentTable;
129  QString _row1("Possible Values;");
130  _row1+=(this->getComboValuesList()).join(COMMA_SEPARATOR);
131  QString _row2("Default Value;");
132  _row2+= this->getDefaultValue();
133  QString _row3("Variable Name in SU2 file;");
134  _row3+=this->getVariableFileName();
135 
136  _contentTable.append(_row1.split(SEMICOLON_SEPARATOR));
137  _contentTable.append(_row2.split(SEMICOLON_SEPARATOR));
138  _contentTable.append(_row3.split(SEMICOLON_SEPARATOR));
139  _htmlFormatter.appendTable(_headerTable,_contentTable);
140  _htmlFormatter.endFormatting();
141  _text = _htmlFormatter.getFormatedText();
142 
143  }else
144  {
145  _text = _helpContents->getDefinitionText();
146  }
147 
148  return _text;
