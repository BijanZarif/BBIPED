36 void MainWindow::launchActionProjectSettingsDialog()
37 {
38  EditHelper _editHelper;
39  bool _userMadeChanges = _editHelper.launchSettingsDialog(_isMultizoneProject,&_multizoneProjectData,&_projectData,&_projectDetailsData);
40  if (_userMadeChanges)
41  {
42  bool _needToRemoveMaps = false;
43  if (!_isMultizoneProject)
44  {
45  // _projectData = _settingsDialog.getProjectData();
46  if (_projectData.getSolverDataInformation().isConfigFileEmpty())
47  {
48  _inputDataTabHelper.updateConfigurationFilePathInActionGroup("");
49  ui->meshFileLineLabel->setText("");
50  _userConfigValuesSolver.remove("MESH_FILENAME");
51 
52  // If we remove the configuration file we MUST remove the associated mesh file and their values
53  if (!_projectData.getSolverDataInformation().isMeshFileEmpty())
54  _projectData.getSolverDataInformation().clear();
55  _needToRemoveMaps = true;
56  }
57  }
58  else
59  {
60  // Refresh the Multizone Project Data
61  _mManager.refreshSimulationFilesView(&_multizoneProjectData);
62  QString _openProject;
63  if (!_projectData.isEmpty())
64  _openProject = _projectData.getSolverDataInformation().getSolverConfigFilePath();
65  QStringList _listConfigFiles = _multizoneProjectData.getListConfigFiles();
66  // We must remove all the data only in the case that the opened project has been removed by the user
67  if (!_listConfigFiles.contains(_openProject))
68  {
69  _needToRemoveMaps = true;
70  // Restore the view names
71  _mManager.updateConfigFileNameinTabView("");
72  _mManager.updateMeshFileNameinTabView("");
73 
74  }
75 
76  }
77  if (_needToRemoveMaps)
78  {
79  _userConfigValuesSolver.clear();
80  resetBoundaryValues();
81  /*_userBoundaryValuesSolver.clear();
82  _boundaryMeshValues.clear();*/
83  // Disable the elements of the menu
84  ui->basivViewEditButton->setEnabled(false);
85  //ui->meshRefreshButton->setEnabled(false);
86  ui->setMeshPropertiesButton->setEnabled(false);
87  ui->savePhysicsButton->setEnabled(false);
88  ui->seeSummaryChangesButton->setEnabled(false);
89  }
90 
91  // Enabling the values
92  setEnableSaveProjectIcons(true);
93  _status_NeedToBeSaved = true;
94  }
95 }
