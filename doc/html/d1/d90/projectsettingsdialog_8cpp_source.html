116 void ProjectSettingsDialog::addValuesToTable(QStringList _list1, QStringList _list2)
117 {
118  bool _isDifSize = false;
119  int _listSize = _list1.size();
120  // Check if the lists are not equal
121  if ((_list1.size() != _list2.size()))
122  _isDifSize = true;
123  if ( (_isDifSize) && (_list1.size()>_list2.size()) )
124  _listSize = _list2.size();
125 
126  for(int i= 0; i < _listSize; i++)
127  {
128  QString _configFileName = _list1.at(i);
129  QString _meshFileName = _list2.at(i);
130  int _cRow = i*2;
131  addRowToTable(_cRow,_cRow/2, _configFileName);
132  _cRow++;
133  addRowToTable(_cRow,_cRow/2, _meshFileName);
134  }
135  // Completing the
136  if (_isDifSize)
137  {
138  bool _addToConfig =false;
139  bool _addToMesh = false;
140  int _difSizeVal = 0;
141  int _startPoint = _listSize;
142  if (_list1.size() > _list2.size() )
143  {
144  _difSizeVal = _list1.size() - _list2.size();
145  _addToConfig = true;
146  }else
147  {
148  _difSizeVal = _list2.size() - _list1.size();
149  _addToMesh = true;
150  }
151  for (int j = _startPoint; j < (_difSizeVal+_startPoint);j++)
152  {
153  QString _configFileName("Not Defined");
154  if (_addToConfig)
155  _configFileName = _list1.at(j);
156  QString _meshFileName("Not Defined");
157  if (_addToMesh)
158  _meshFileName = _list2.at(j);
159  int _cRow = j*2;
160  addRowToTable(_cRow,_cRow/2, _configFileName);
161  _cRow++;
162  addRowToTable(_cRow,_cRow/2, _meshFileName);
163  }
164 
165  }

199 void ProjectSettingsDialog::updateProjectData()
200 {
201  if (!_projectData.isEmpty())
202  {
203  SolverVO _solverData = _projectData.getSolverDataInformation();
204  QString _configFileName = QFileInfo(_solverData.getSolverConfigFilePath()).fileName();
205  QString _meshFileName = QFileInfo(_solverData.getSolverFilePath()).fileName();
206  if ( (QString::compare(_configFileName,_removedConfigFiles.at(0)) == StrEqual)
207  && (QString::compare(_meshFileName,_removedMeshFiles.at(0)) == StrEqual) )
208  _solverData.clear();
209  _projectData.setSolverDataInformation(_solverData);
210 
211  }else
212  {
213  if (!_multizoneData.isEmpty())
214  {
215  QStringList _configFileList = _multizoneData.getListConfigFiles();
216  QStringList _configFileListNames = _multizoneData.getListConfigFileNames();
217  QStringList _meshFileList = _multizoneData.getListMeshFiles();
218  QStringList _meshFileListNames = _multizoneData.getListMeshFileNames();
219 
220  for (int i = 0; i < _removedConfigFiles.size();i++)
221  {
222  int _posToRemove = _configFileListNames.indexOf(_removedConfigFiles.at(i));
223  _configFileList.removeAt(_posToRemove);
224  _configFileListNames.removeAt(_posToRemove);
225  }
226 
227  for (int i = 0; i < _removedMeshFiles.size();i++)
228  {
229  int _posToRemove = _meshFileListNames.indexOf(_removedMeshFiles.at(i));
230  _meshFileList.removeAt(_posToRemove);
231  _meshFileListNames.removeAt(_posToRemove);
232  }
233  _multizoneData.setListConfigFiles(_configFileList);
234  _multizoneData.setListMeshFiles(_meshFileList);
235  }
236 
237  }
