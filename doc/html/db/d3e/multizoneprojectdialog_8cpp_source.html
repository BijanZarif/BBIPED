177 bool MultizoneProjectDialog::buildFilesInProject()
178 {
179  UtilFiles _fileHelper;
180  int _rowConfigFile;
181  int _rowMeshFile;
182  bool _buildSuccess = true;
183  QString _fileConfigName;
184  QString _fileMeshName;
185  SolverConfigManager _solverManager;
186  QHash<QString,QString> _userVal;
187 
188  QStringList _listConfigFiles;
189  QStringList _listMeshFiles;
190 
191  if (!_solverManager.isInitialized())
192  _solverManager.initializeManager();
193 
194  int _lastRow = ui->configTable->rowCount();
195  int _processedRow = 0;
196 
197 
198  while(_processedRow < _lastRow )
199  {
200  _rowConfigFile = _processedRow;
201  _rowMeshFile = _processedRow+1;
202 
203  _fileConfigName = ui->configTable->item(_rowConfigFile, ColumnFileName)->text();
204  _fileMeshName = ui->configTable->item(_rowMeshFile, ColumnFileName)->text();
205 
206  if (!isMeshFileNameValid(_fileMeshName))
207  {
208  ShowMessageHelper _messageHelper;
209  bool _result = _messageHelper.showWarningMessage("Mesh File Name not Valid", "Please, choose a valid mesh file to continue");
210  return (!_buildSuccess);
211  }
212  else{
213  // Import Mesh File
214  QString _newMeshPath = _fileHelper.importFile(_projectPath,_fileMeshName, TOOL4);
215  _listMeshFiles.append(_newMeshPath);
216  _userVal.insert("MESH_FILENAME",_newMeshPath);
217  // Process the Config file
218  QString _finalPath = QDir(_projectPath).filePath(TOOL4);
219  if (_newFilesList.contains(_fileConfigName))
220  {
221  // Saving with all the values by default and the mesh file name associated to it
222  _solverManager.saveConfigFile(QDir(_finalPath).filePath(_fileConfigName),_userVal, _projectDetailsData,_virtualMRFData);
223  _listConfigFiles.append(QDir(_finalPath).filePath(_fileConfigName));
224  }else
225  {
226  QString _newPath;
227  if (_importedFilesList.contains(_fileConfigName))
228  {
229  _newPath = _fileHelper.importFile(_projectPath,_fileConfigName, TOOL4);
230  bool _canContinue = checkConsistency(_newPath, _newMeshPath, _solverManager);
231  if (!_canContinue)
232  {
233  _buildSuccess = false;
234  return _buildSuccess;
235  }
236 
237  }else // The user has not done anything --> Create a new one with default name
238  {
239  _newPath = QDir(_finalPath).filePath("default_config_"+QString::number(_rowConfigFile)+".cfg");
240  _solverManager.saveConfigFile(_newPath,_userVal,_projectDetailsData,_virtualMRFData);
241  }
242  _listConfigFiles.append(_newPath);
243  }
244  // Update the counter
245  _processedRow = _processedRow+2;
246  _userVal.clear();
247  }
248  _multizoneProjectData.setListConfigFiles(_listConfigFiles);
249  _multizoneProjectData.setListMeshFiles(_listMeshFiles);
250 
251  }
252  return _buildSuccess;
253 }

522  _isConsistent = false;
523  else{

525  {

527  _userConfigVal.insert("MESH_FILENAME",_meshFilePath);
528  }
529  // Saving the file
530  _solverManager.saveConfigFile(_configFilePath,_userConfigVal,_projectDetailsData,_virtualMRFData);
531  }
532  }
533  }
534  return _isConsistent;
