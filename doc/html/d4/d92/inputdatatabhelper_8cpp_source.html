78 void InputDataTabHelper::parseFile(int _option,QString _pathFile,
79  QHash<QString,QString> * _userConfigVal,
80  QHash<QString,QString> * _boundaryMeshVal,
81  QList<VirtualMrfVO> * _virtualVOList)
82 {
83  switch(_option)
84  {
85 
86  case ParserSolverConfigFile:
87  {
88  ConfigurationFileParser _configParser;
89  _configParser.setFilePath(_pathFile);
90  _configParser.parseFileProcess();
91  *_userConfigVal = _configParser.getParsedValues();
92  * _virtualVOList = _configParser.getParsedVirtualMRFData();
93  }
94  break;
95  case ParserSolverMeshFile:
96  {
97  MeshFileParser _meshParser;
98  _meshParser.setFilePath(_pathFile);
99  _meshParser.parseFileProcess();
100  *_boundaryMeshVal = _meshParser.getParsedValues();
101  bool _isMesh3D = _meshParser.is3DMesh();
102  if (_isMesh3D)
103  *_boundaryMeshVal->insert(IS_3D_MESH_TAG,"YES");
104  }
105  break;
106  }
107 }

188 bool InputDataTabHelper::updateSolverUserValues(ProjectData _projectData,
189  QHash<QString,QString> * _userConfigVal,
190  QHash<QString,QString> * _boundaryMeshVal,
191  QHash<QString,QString> * _boundaryConfigVal,
192  QList<VirtualMrfVO> * _virtualVOList)
193 {
194 
195  SolverVO _solverVO = _projectData.getSolverDataInformation();
196  if (!_solverVO.getSolverConfigFilePath().isEmpty())
197  parseFile(ParserSolverConfigFile,_solverVO.getSolverConfigFilePath(),_userConfigVal,
198  _boundaryMeshVal,_virtualVOList);
199  if (!_solverVO.getSolverFilePath().isEmpty())
200  parseFile(ParserSolverMeshFile,_solverVO.getSolverFilePath(),_userConfigVal,_boundaryMeshVal,_virtualVOList);
201 
202  return false;
203 }
