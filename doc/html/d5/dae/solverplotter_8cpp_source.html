204 void SolverPlotter::showPoints(SolverResidualVO * _obj)
205 {
206  int _iteration = _obj->getIteration();
207  updateLegendItemsInGraphic(_obj);
208  if (_iteration < 0 )
209  return;
210  // Timing
211  if (_obj->getResRhoEnabled())
212  {
213  data[resRho]._points << QPointF(_iteration,_obj->getResRho());
214  data[resRho].curve->setSamples( data[resRho]._points);
215  }
216  if (_obj->getResRhoeEnabled())
217  {
218  data[resRhoe]._points << QPointF(_iteration,_obj->getResRhoe());
219  data[resRhoe].curve->setSamples( data[resRhoe]._points);
220  }
221 
222  if( (_obj->getResRhoe() > 50) || (_obj->getResRhoe() < -50) ||
223  (_obj->getResRho() > 50) || (_obj->getResRho() < -50) )
224  setAxisAutoScale(QwtPlot::yLeft,true);
225  if (_iteration > 100)
226  setAxisAutoScale(QwtPlot::xBottom,true);
227  replot();
228 }
