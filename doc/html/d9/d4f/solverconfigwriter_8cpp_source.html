197 void SolverConfigWriter::writeSubsectionElement(QDomElement _subSectionElement, QTextStream * _fileOutput)
198 {
199  if (!_subSectionElement.isNull())
200  {
201  QDomNodeList _childList = _subSectionElement.childNodes();
202  bool _isVirtualMRF = false;
203 
204  if (!_childList.isEmpty())
205  for (int i = 0;i < _childList.count();i++)
206  {
207  QDomElement _subsectionEl = _childList.at(i).toElement();
208  if (!_subsectionEl.isNull())
209  {
210  if (QString::compare(_subsectionEl.tagName(),XML_SUBSECTION_TITLE_NAME) == StrEqual)
211  {
212  *_fileOutput << "%%--------------------------------------------------------------------------------------" << endl;
213  *_fileOutput << "%% " << _subsectionEl.text() << endl;
214  *_fileOutput << "%%--------------------------------------------------------------------------------------" << endl;
215  // If is Virtual MRF section a customized output must be done
216  if (QString::compare(_subsectionEl.text(),VIRTUAL_SUBSECTION_NAME) == StrEqual)
217  _isVirtualMRF = true;
218  }
219  if (QString::compare(_subsectionEl.tagName(),XML_VARIABLE_LIST_NAME) == StrEqual)
220  writeVariableList(_subsectionEl,_fileOutput,_isVirtualMRF);
221  }
222  }
223 
224  // In the case ov Virtual MRF -- Add hidden Variables
225  if ((_isVirtualMRF) && (!_virtualMRFUserValues.isEmpty()))
226  {
227  // Write the hidden Variables
228  QString _value;
229  _value=VMRF_GRID_MOVEMENT_KEYNAME+"=(YES)";
230  _value.simplified();
231  _value.replace(" ","");
232  *_fileOutput << _value << endl;
233  _value=VMRF_GRID_MOVEMENT_KIND_KEYNAME+"=(ROTATING_FRAME)";
234  _value.simplified();
235  _value.replace(" ","");
236  *_fileOutput << _value << endl;

238  }

300 void SolverConfigWriter::writeVariableValue(QDomElement _valueElement, QTextStream * _fileOutput,bool _isVirtualMRF)
301 {
302  QString _value;
303  if (_valueElement.hasAttribute(XML_VARIABLE_VALUE_NAME_ATT)) {
304 
305  if (!_isVirtualMRF)
306  {
307 
308  QString _keyValue = _valueElement.attribute(XML_VARIABLE_VALUE_NAME_ATT);
309  bool _saveValue = true;
310  if ((!_optionalVariables.isEmpty()) && (_optionalVariables.contains(_keyValue)))
311  _saveValue = false;
312 
313  if ((!_userValues.isEmpty()) && (_userValues.contains(_keyValue)))
314  {
315  _value = _valueElement.attribute(XML_VARIABLE_VALUE_NAME_ATT);
316  QString _cValue = _userValues.value(_keyValue);
317  if (QString::compare(_keyValue,DOMAIN_CLIPPING_VARIABLE_FILENAME) == StrEqual)
318  {
319  if (_cValue.isEmpty())
320  _value ="";
321  else
322  {
323  // Print each value in several Domain clip variables in the file
324  QStringList _valueList = _cValue.split(";");
325  for (int i =0; i < _valueList.size();i++)
326  {
327  _value = DOMAIN_CLIPPING_VARIABLE_FILENAME+ "=("+_valueList.at(i)+")";
328  _value.simplified();
329  _value.replace(" ","");
330  *_fileOutput << _value << endl;
331  _value="";
332  }
333  }
334 
335  }else
336  {
337  if ( ((_value.contains(MARKER_PREFIX)) || (_value.contains(BC_PREFIX)))
338  && (!_cValue.contains("(")) && (!_cValue.contains(")")))
339  _value = _value +"= ("+ _cValue+ ")";
340  else
341  _value = _value +"="+ _cValue;
342  }
343  }else
344  if (_saveValue)
345  _value = _valueElement.attribute(XML_VARIABLE_VALUE_NAME_ATT) +"="+_valueElement.text();
346 
347  _value.simplified();
348  _value.replace(" ","");
349  *_fileOutput << _value << endl;
350  }else
351  printVirtualMRFValues(_valueElement,_fileOutput);
352  }
