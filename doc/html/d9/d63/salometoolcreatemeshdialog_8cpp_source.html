83 void SalomeToolCreateMeshDialog::runPythonFile()
84 {
85  QFileInfo _salomeFilePath(_salomeMeshPath);
86  QString _salomeToolPath = _salomeFilePath.absolutePath();
87  if (_salomeToolPath.isEmpty())
88  return;
89 
90  _regEditor->append("Starting Python JobManager Execution ....");
91  _meshCreationProcess = new QProcess(this);
92  _meshCreationProcess->setWorkingDirectory(QDir(_projectPath).filePath("Python"));
93  QString _meshCommand("") ;
94  _meshCommand = _salomeToolPath + "/runSession runSalome -t -u" + _projectName +".py";
95  connect(_meshCreationProcess, SIGNAL(error(QProcess::ProcessError)),
96  this, SLOT(slotProcessError(QProcess::ProcessError)));
97  connect(_meshCreationProcess, SIGNAL(readyReadStandardOutput()), this, SLOT(outputConsole()));
98  connect(_meshCreationProcess, SIGNAL(readyReadStandardError()), this, SLOT(errorConsole()));
99  connect(_meshCreationProcess, SIGNAL(finished(int)), this, SLOT(meshCreationFinished()));
100 
101  _regEditor->append("[START MESH CREATION] --> Running " + _meshCommand);
102  _meshCreationProcess->start(_meshCommand);
103 
104  // Updating the Text of the Create Mesh Button
105  ui->createMeshButton->setDisabled(true);
106 }
