194 void WarningMultizoneDialog::fillRepairTable()
195 {
196  QHash<QString,QString> _oldMeshValues = getMarkerValues();
197  QStringList _meshVariableName = getMeshVarNames();
198 
199  if (!_oldMeshValues.isEmpty())
200  {
201  QHashIterator<QString,QString> iter(_oldMeshValues);
202  SolverSectionDAO _dao;
203  while(iter.hasNext())
204  {
205  iter.next();
206  SolverVariableVO _varVO = _dao.findVariableByFileName(_menuSectionList,iter.key());
207  if ( (!_varVO.isEmpty()) && (!iter.value().contains("NONE")) )
208  buildRow(_varVO.getVariableName(),iter.value(),_meshVariableName,_varVO.getVariableFileName());
209  }
210  connect(_signalMapper, SIGNAL(mapped(const QString &)),this, SLOT(changedComboValueByUser(const QString &)),Qt::UniqueConnection);
211  _repairTableFilled = true;
212  }

223 void WarningMultizoneDialog::buildRow(QString _markerFileName,QString _cValue,QStringList _meshVarNameList,QString _markerKeyName)
224 {

226  int _lastRow = ui->repairMeshValuesTable->rowCount();
