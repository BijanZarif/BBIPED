93 void MultizoneIndividualTabForm::on_importMeshFileButton_clicked()
94 {
95 
96  if (!_meshFile.isEmpty())
97  {
98  ShowMessageHelper _msgHelper;
99  bool _overWrite = _msgHelper.showWarningMessage("Mesh File Overwrite!","If you import a new file, you will loose all your previous configuration. Do you want to continue?");
100  if (!_overWrite)
101  return;
102  }
103  QString _meshFilePath = QFileDialog::getOpenFileName(_parentWindow,"Import SU2 Mesh File",
104  QDir(_projectPath).filePath(TOOL4),
105  "SU2 Mesh File (*.su2)");
106  if (_meshFilePath.isNull())
107  return;
108 
109  // Step 1. Import the mesh File into the project folder
110  UtilFiles _utilHelper;
111  _meshFile = _utilHelper.importFile(QFileInfo(_projectPath).path(),_meshFilePath,TOOL4);
112 
113  // Step 2. Reset previous Bnd Values
114  _bndMeshVal->clear();
115  _bndValuesSolver->clear();
116  removeAssociatedMeshValues();
117 
118  // Step 2. Parse the mesh file to get the bnd Variables
119  _regTabHelper->parseFile(ParserSolverMeshFile,_meshFile,_configVal,_bndMeshVal,_virtualMRFList);
120  _configVal->insert("MESH_FILENAME",_meshFile);
121  // Step 4. Update the view
122  ui->currentMeshFileLabel->setText(QFileInfo(_meshFile).fileName());
123  _needToSave = true;
124  ui->saveButton->setEnabled(true);
125  addSaveIconTab(iconSaveSepia);
126 }
