67 void SolverConfigManager::loadConfigTemplate()
68 {
69  // Loading the file from the resource
70  QFile * _templateConfigFile = new QFile(CONFIG_TEMPLATE_PATH);
71  _configDomTree = new QDomDocument();
72 
73  if( !_templateConfigFile->open( QIODevice::ReadOnly | QIODevice::Text ) )
74  {
75  qDebug( "Failed to open file for reading." );
76  }
77  else
78  {
79  if( !_configDomTree->setContent( _templateConfigFile ) )
80  {
81  qDebug( "Failed to parse the file into a DOM tree." );
82  _templateConfigFile->close();
83  }
84  _templateConfigFile->close();
85  buildBasicMenuList();
86  buildAdvancedMenuList();
87  buildMenuVariablesList();
88  buildBoundaryMenuList();
89  buildCompleteMenuList();
90  _isInitialized = true;
91  }
92  _templateConfigFile->close();
93 
94 }

294 void SolverConfigManager::updateVisibility(QStringList _listVariables, bool _visibility)
295 {
296  if ((!_listVariables.isEmpty()) && (!_sectionMenuList.isEmpty()))
297  {
298 
299  for(int i = 0; i < _sectionMenuList.size();i++)
300  {
301  SolverSectionVO _section = _sectionMenuList.at(i);
302  QList<SolverVariableVO> _varList = _section.getSubsectionVarList();
303  QList<SolverVariableVO> _newVarList;
304  if (!_varList.isEmpty())
305  for (int j= 0; j<_varList.size();j++)
306  {
307  SolverVariableVO _var = _varList.at(j);
308  if (_listVariables.contains(_var.getVariableFileName()))
309  _var.setVisibility(_visibility);
310  _newVarList.append(_var);
311  }
312  _sectionMenuList[i].setSubsectionVarList(_newVarList);
313  }
314  }
315 }
