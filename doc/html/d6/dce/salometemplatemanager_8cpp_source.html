168 void SalomeTemplateManager::buildTemplateData(QDomElement _templateNode,int _option)
169 {
170  if ( (!_templateNode.isNull()) && (QString::compare(_templateNode.tagName(),XML_TEMPLATE_NODE) == StrEqual))
171  {
172  QDomNodeList _templateChildren= _templateNode.childNodes();
173  SalomeTemplateData _templateData;
174  for (int i = 0; i < _templateChildren.size(); i++)
175  {
176  QDomElement _templateEl = _templateChildren.at(i).toElement();
177 
178  if ((_templateEl.isNull()) || (_templateEl.tagName().isEmpty()) )
179  continue;
180  if (QString::compare(_templateEl.tagName(),XML_TEMPLATE_NAME_NODE) == StrEqual)
181  _templateData.setTemplateName(_templateEl.text());
182 
183  if (QString::compare(_templateEl.tagName(),XML_TEMPLATE_FILE_NAME_NODE) == StrEqual)
184  _templateData.setTemplateFileName(_templateEl.text());
185  if (QString::compare(_templateEl.tagName(),XML_VARIABLE_LIST_NODE) == StrEqual)
186  buildVariableList(_templateEl,&_templateData);
187 
188  }
189  // Saving the data of the template
190  if (_option == TEMPLATE_LIST)
191  {
192  _templatesList.push_back(_templateData);
193  _templateNameList.push_back(_templateData.getTemplateName());
194  }
195  else
196  {
197  _userTemplatesList.push_back(_templateData);
198  _userTemplateNameList.push_back(_templateData.getTemplateName());
199  }
200  }
