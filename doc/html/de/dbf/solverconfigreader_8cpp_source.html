80 QList<SolverVariableVO> SolverConfigReader::createVariableSection(QDomElement _element)
81 {
82  QList<SolverVariableVO> _resultList;
83  QDomNodeList _variableList = _element.elementsByTagName(XML_VARIABLE_NAME);
84  if (!_variableList.isEmpty())
85  {
86  for (int i = 0;i < _variableList.count();i++)
87  {
88  SolverVariableVO _createdObj;
89  QDomElement _child = _variableList.at(i).toElement();
90  // Check if it is repeatable
91  if (_child.hasAttribute(XML_VARIABLE_ISREPEATABLE_ATT))
92  if (QString::compare(_child.attribute(XML_VARIABLE_ISREPEATABLE_ATT),YES_VALUE) == StrEqual)
93  _createdObj.setRepeatable(true);
94  // Check if it is optional
95  if (_child.hasAttribute(XML_VARIABLE_ISOPTIONAL_ATT))
96  if (QString::compare(_child.attribute(XML_VARIABLE_ISOPTIONAL_ATT),YES_VALUE) == StrEqual)
97  _createdObj.setOptional(true);
98  // Check if it is selectable by the users
99  if (_child.hasAttribute(XML_VARIABLE_SELECTABLE_ATT))
100  if (QString::compare(_child.attribute(XML_VARIABLE_SELECTABLE_ATT),"YES") == StrEqual)
101  _createdObj.setVariableSelectable(true);
102  else
103  _createdObj.setVariableSelectable(false);
104 
105  QDomNodeList _nameList = _child.elementsByTagName(XML_VARIABLE_NAME_NAME);
106 
107  // Variable Name
108  if (!_nameList.isEmpty())
109  _createdObj.setVariableName(_nameList.at(0).toElement().text());
110  QDomNodeList _helpList = _child.elementsByTagName(XML_VARIABLE_HELP_NAME);
111  // Help Value
112  if (!_helpList.isEmpty())
113  _createdObj.setHelpContents(createHelpElement(_helpList.at(0).toElement()));
114 
115  QDomNodeList _valueList = _child.elementsByTagName(XML_VARIABLE_VALUE_NAME);
116  if (!_valueList.isEmpty())
117  {
118  QDomElement _valueEl = _valueList.at(0).toElement();
119  // Variable File Name
120  if (_valueEl.hasAttribute(XML_VARIABLE_VALUE_NAME_ATT))
121  _createdObj.setVariableFileName(_valueEl.attribute(XML_VARIABLE_VALUE_NAME_ATT));
122  // Default Value
123  if (_valueEl.hasAttribute(XML_VARIABLE_VALUE_DEFAULT_VALUE_ATT))
124  {
125  _createdObj.setDefaultValue(_valueEl.attribute(XML_VARIABLE_VALUE_DEFAULT_VALUE_ATT));
126  _createdObj.setCurrentValue(_valueEl.text());
127  }
128  // Unit Value
129  if (_valueEl.hasAttribute(XML_VARIABLE_VALUE_UNIT_ATT))
130  {
131  QString _unitValue = _valueEl.attribute(XML_VARIABLE_VALUE_UNIT_ATT);
132  if (QString::compare(_unitValue,"NONE") == StrEqual) // If there is no unit applicable, we don't shown anything
133  _unitValue = "";
134  _createdObj.setUnitValue(_unitValue);
135  }
136  // In case it has Other Variable Names inside
137  if (_valueEl.hasAttribute(XML_VARIABLE_VALUE_SUBVAR_NAME_ATT))
138  {
139  QString _subVariableNameList = _valueEl.attribute(XML_VARIABLE_VALUE_SUBVAR_NAME_ATT);
140  _createdObj.setSubVariableNames(_subVariableNameList.split(","));
141  }
142  // Check if it is Visible
143  if (_valueEl.hasAttribute(XML_VARIABLE_ISVISIBLEINMENU_ATT))
144  if (QString::compare(_valueEl.attribute(XML_VARIABLE_ISVISIBLEINMENU_ATT),"NO") == StrEqual)
145  _createdObj.setVisibleInMenu(false);
146  // Check if it contains associated variables
147  if (_valueEl.hasAttribute(XML_VARIABLE_ASSOCIATED_VAR_ATT))
148  {
149  QString _associatedVarList = _valueEl.attribute(XML_VARIABLE_ASSOCIATED_VAR_ATT);
150  _createdObj.setAssociatedVariables(_associatedVarList.split(","));
151  }
152  }
153 
154  // Combo Values
155  QDomNodeList _comboValuesParent = _child.elementsByTagName(XML_VARIABLE_COMBOVALUES_LIST_NAME);
156  if (! _comboValuesParent.isEmpty())
157  {
158  QDomElement _comboAttElement = _comboValuesParent.at(0).toElement();
159  QString _uniqueValue;
160  if (_comboAttElement.hasAttribute(XML_VARIABLE_COMBOVALUE_UNIQUESEL_ATT))
161  _uniqueValue = _comboAttElement.attribute(XML_VARIABLE_COMBOVALUE_UNIQUESEL_ATT);
162 
163  // If the combo values has selected no unique option, this implies that it allows multiple
164  // selection (similar to checkbox
165  if ((!_uniqueValue.isEmpty()) && (QString::compare(_uniqueValue,"YES") != StrEqual))
166  _createdObj.setUniqueSelection(false);
167  }
168  createComboValues(_child, &_createdObj);
169 
170  _resultList.append(_createdObj);
171 
172  }
173  }
174  return _resultList;
175 }
