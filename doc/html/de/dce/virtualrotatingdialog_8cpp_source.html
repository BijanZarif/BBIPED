253 void VirtualRotatingDialog::createVirtualMRFData(int _row)
254 {
255  int _totalRows = ui->zoneInfoTable->rowCount();
256  int _totalObjects = _resultUserVirtualMRFList.count();
257  bool _needToAdd = false;
258 
259  for (int i = 0; i < _totalRows; i++)
260  {
261  VirtualMrfVO _virtualInfo;
262  if (i >= _totalObjects)
263  _needToAdd = true;
264  else
265  _virtualInfo = _resultUserVirtualMRFList.at(i);
266 
267  if (i == _row)
268  {
269  _virtualInfo.setZoneName(((QLineEdit *) ui->zoneInfoTable->cellWidget(_row,ColumnZoneName))->text());
270  _virtualInfo.setOmega(((QLineEdit *) ui->zoneInfoTable->cellWidget(_row,ColumnOmega))->text());
271 
272  QString _centerValue = ((QLineEdit *) ui->zoneInfoTable->cellWidget(_row,ColumnAxisCenter))->text();
273 
274  if (_is2DMesh)
275  {
276  QStringList _centerValueList = _centerValue.split(",");
277  if (_centerValueList.size() > 2)
278  for (int i=_centerValueList.size(); i >= 2;i--)
279  _centerValueList.removeAt(i);
280  _centerValueList.append("0");
281  _centerValue = _centerValueList.join(",");
282  }
283 
284  _virtualInfo.setAxisCenter(_centerValue.split(","));
285  if (_is2DMesh)
286  _virtualInfo.setAxisNormal(QString("0,0,1").split(","));
287  else
288  _virtualInfo.setAxisNormal(((QLineEdit *) ui->zoneInfoTable->cellWidget(_row,ColumnAxisNormal))->text().split(","));
289 
290  }
291  if (_needToAdd)
292  _resultUserVirtualMRFList.append(_virtualInfo);
293  else
294  _resultUserVirtualMRFList[i] = _virtualInfo;
295  }

326 void VirtualRotatingDialog::launchCurveDialog(int _row)
327 {
328  VirtualMrfVO _virtualVO;
329  _virtualVO = _resultUserVirtualMRFList.at(_row);
330 
331 
332  VirtualCurveDialog _virtualCurveDialog;
333  _virtualCurveDialog.setSolverSection(_solverSection);
334 
335  if (!_virtualVO.isEmpty())
336  {
337  QStringList _listInnerPoints = _virtualVO.getInnerCurvePoints();
338  QStringList _listOuterPoints = _virtualVO.getOuterCurvePoints();
339  QString _point1;
340  QString _point2;
341 
342  if (_is2DMesh)
343  {
344  _point1 = "-1,0";
345  _point2 = "1,0";
346  }else
347  {
348  // By default in 3D, we need a minimum of 2 points
349  _point1 = "0,0";
350  _point2 = "0,0";
351  }
352  if (_listInnerPoints.size() == 0)
353  {
354  _listInnerPoints.append(_point1.split(","));
355  _listInnerPoints.append(_point2.split(","));
356  }
357  if (_listOuterPoints.size() == 0)
358  {
359  _listOuterPoints.append(_point1.split(","));
360  _listOuterPoints.append(_point2.split(","));
361  }
362  if ((_is2DMesh) && (_listInnerPoints.size() > 4 ))
363  for (int i=_listInnerPoints.size();i>3;i--)
364  _listInnerPoints.removeAt(i);
365  // we need to be sure that the points follow the same structre 1,x; -1,y
366  _listInnerPoints.replace(0, QString("-1"));
367  _listInnerPoints.replace(2, QString("1"));
368 
369 
370  if ((_is2DMesh) && (_listOuterPoints.size() > 4 ))
371  for (int i=_listOuterPoints.size();i>3;i--)
372  _listOuterPoints.removeAt(i);
373 
374  // we need to be sure that the points follow the same structre 1,x; -1,y
375  _listOuterPoints.replace(0, QString("-1"));
376  _listOuterPoints.replace(2, QString("1"));
377 
378  _virtualCurveDialog.setInnerValues(_listInnerPoints);
379  _virtualCurveDialog.setOuterValues(_listOuterPoints);
380  _virtualCurveDialog.set3DCoordinates(!_is2DMesh);
381  _virtualCurveDialog.initializeDialog();
382  }
383  _virtualCurveDialog.exec();
384 
385  // If it is accepted the changes we can mark is as done
386  if (_virtualCurveDialog.hasUserModifiedValues())
387  {
388  ui->zoneInfoTable->item(_row,ColumnDone)->setCheckState(Qt::Checked);
389  // Save the results of the user
390  _virtualVO.setInnerCurvePoints(_virtualCurveDialog.getInnerValues());
391  _virtualVO.setOuterCurvePoints(_virtualCurveDialog.getOuterValues());
392  _virtualVO.setCoordinatesIn3D(_virtualCurveDialog.is3DCoordinates());
393  _resultUserVirtualMRFList.replace(_row,_virtualVO);
394 
395  }
396
